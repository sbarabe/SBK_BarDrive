251 void setPixel(uint8_t segment, uint8_t state)
252 {
253 if (segment >= _numSegments)
254 return;
255 if (_direction == BarDirection::REVERSE)
256 segment = (_numSegments - 1) - segment;
257
258 uint8_t row, col;
259 _getMappedRowCol(segment, &row, &col);
260 _driver->setLed(_deviceIndex, row, col, state != 0);

281 uint8_t getPixelState(uint8_t segment) const
282 {
283 if (segment >= _numSegments)
284 return false;
285
286#ifdef SBK_TRACK_STATE_IN_RAM
287 if (_direction == BarDirection::REVERSE)
288 segment = (_numSegments - 1) - segment;
289
290 return _pixelStateCache ? _pixelStateCache[segment] : false;
291#else
292 if (!_driver)
293 return false;
294
295 uint8_t row, col;
296 _getMappedRowCol(segment, &row, &col);
297 return _driver->getLed(_deviceIndex, row, col);
298#endif

301private:
302 void _initializeMapping(BarMeterType type)
303 {
304 if (type == BarMeterType::SBK_BarMeter_SK28)
305 type = BarMeterType::BL28_3005SK;
306 if (type == BarMeterType::SBK_BarMeter_SA28)
307 type = BarMeterType::BL28_3005SA;
308
309 switch (type)
310 {
311 case BarMeterType::DEFAULT_BarMeter:
312 _rowsNum = 8;
313 _colsNum = 8;
314 break;
315 case BarMeterType::BL28_3005SK:
316 _numSegments = 28;
317 _rowsNum = 7;
318 _colsNum = 4;
319 break;
320 case BarMeterType::BL28_3005SA:
321 _numSegments = 28;
322 _rowsNum = 4;
323 _colsNum = 7;
324 break;
325 case BarMeterType::SBK_BarMeter_24:
326 _numSegments = 24;
327 _rowsNum = 3;
328 _colsNum = 8;
329 break;
330 case BarMeterType::SBK_BarMeter_40:
331 _numSegments = 40;
332 _rowsNum = 5;
333 _colsNum = 8;
334 break;
335 case BarMeterType::CUSTOM:
336 break;
337 default:
338 _numSegments = 64;
339 _rowsNum = 8;
340 _colsNum = 8;
341 break;
342 }
343
344// Allocate pixel state cache
345#ifdef SBK_TRACK_STATE_IN_RAM
346 if (_pixelStateCache)
347 delete[] _pixelStateCache;
348 _pixelStateCache = new bool[_numSegments]{};
349#endif
350 }

352 void _getMappedRowCol(uint8_t index, uint8_t *row, uint8_t *col) const
353 {
354 uint8_t r, c;
355 if (_type == BarMeterType::CUSTOM && _customMapping)
356 {
357 if (_userMappingIsProgmem)
358 {
359 r = pgm_read_byte(&_customMapping[index][0]);
360 c = pgm_read_byte(&_customMapping[index][1]);
361 }
362 else
363 {
364 r = _customMapping[index][0];
365 c = _customMapping[index][1];
366 }
367 }
368 else
369 {
370 if (_direction == BarDirection::REVERSE)
371 index = (_numSegments - 1) - index;
372 r = index / _colsNum;
373 c = index % _colsNum;
374 }
375 *row = r;
376 *col = c;
377 }

386 uint8_t _colsNum = 0;
387 bool _userMappingIsProgmem = false;
388#ifdef SBK_TRACK_STATE_IN_RAM
389 bool *_pixelStateCache = nullptr;
